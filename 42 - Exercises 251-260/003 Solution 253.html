    ''': '.----.',
    ''': '.-..-.',
    '?': '..--..',
    '/': '-..-.',
    '=': '-...-',
    '+': '.-.-.',
    '-': '-....-',
    '(': '-.--.',
    ')': '-.--.-',
    '!': '-.-.--',
}

MORSE_CODE_REVERSED = {val: key for key, val in MORSE_CODE.items()}


def encrypt(message):
    return ' '.join([MORSE_CODE[char.upper()] if char != ' ' else '/' for char in message])


def decrypt(message):
    decipher = ''
    for char in message.split():
        if char != '/':
            decipher += MORSE_CODE_REVERSED[char]
        else:
            decipher += ' '
    return decipher
